#### 节流，防抖

节流：只执行了第一次的

```js
function thr(fn, delay){
	var timer = null;
	return function(){
        if(timer){
            return false;
        }
        var that = this;
        var args = arguments;
        fn.apply(that, args);
        timer = setTimeOut(function(){
            clearTimeOut(timer);
            timer = null;
        }, delay || 500);
    };
}

// 测试
function clickHandle(){
    console.log('节流click')
}

const handler = thr(clickHandler());
document.getElementById("button").addEventListener('click', handler);
```
